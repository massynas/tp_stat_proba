update.variance <- function(var_1, x){
  # Votre code 
  t <- length(x)
  m <- mean(x)
  vx <-  (t-1) * var(x) / t
  new_variance = (t * vx) / (t+1) + t * ((var_1 - m) /(t+1))^2
  return(new_variance)
}
